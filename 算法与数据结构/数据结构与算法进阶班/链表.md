# 链表

## 特性

（单链表）
- 在内存中是分散的
- 要多少给多少，不会有剩余空出来的的空间
- 不能索引访问，只能从头结点一个一个 .next 访问到目标节点
- 在内存中保存头结点

### 自己构造链表

```python
# from util.Empty import Empty
# from util.Outbound import Outbound

# 链表节点类
class Node:
    def __init__(self, value=None, next=None):
        self.value = value
        self.next = next
    
# 链表类
class LinkedList:
    def __init__(self):
        self.head = Node()  # 哨兵结点
        self.length = 0     # 链表长度要有
    
    def peek(self):
        """找到头结点（因为第一个是哨兵结点）"""
        if not self.head.next:
            raise ValueError('LinkedList id empty')
        return self.head.next
    
    def add_first(self, value):
        """在最前面插入结点"""
        # 其实是在哨兵结点和头结点之间插入结点
        new_node = Node(value, None)
        new_node.next = self.head.next  
        self.head.next = new_node
        self.length += 1
    
    def add_last(self, value):
        new_node = Node(value, None)
        node = self.head
        while node.next != None:
            node = node.next
        node.next = new_node    # new_node 的 next 为空（尾结点）
        self.length += 1
    
    def add(self, index, value):
        if (index < 0 or index >= self.length):
            raise Outbound('index is out of bound')
        if not self.head.next:
            raise ValueError('LinkedList id empty')
        
        node = self.head
        for _ in range(index):
            node = node.next
        # 此时 node 为 index 索引上的结点
        new_node = Node(value, None)
        new_node.next = node.next
        node.next = new_node
        self.length += 1

    def get_first(self):
        """找到头结点 O(1)"""
        if not self.head.next:
            raise ValueError('LinkedList id empty')
        return self.head.next
    
    def get_last(self):
        """找到最后一个结点 O(n)"""
        if not self.head.next:
            raise ValueError('LinkedList id empty')
        node = self.head
        while node.next != None:
            node = node.next
        return node
    
    def get(self, index):
        """找到指定索引的结点"""
        if (index < 0 and  index >= self.length):
            raise ValueError('index is out of bound')
        if not self.head.next:
            raise ValueError('LinkedList id empty')

        node = self.head.next
        for _ in range(index):
            node = node.next
        return node

    def remove_first(self):
        """删除头结点"""
        if not self.head.next:
            raise ValueError('LinkedList id empty')
        re_node = self.head.next
        self.head.next = self.head.next.next
        self.length -= 1
        return re_node.value      # 返回被删除的头结点的值

    def remove_last(self):
        """删除最后一个结点"""
        if not self.head.next:
            raise ValueError('LinkedList id empty')
        node_pre = self.head
        node = self.head.next
        while node.next != None:
            node_pre = node
            node = node.next
        # 此时 node 为最后一个结点, node_pre 为最后一个结点的前一个结点
        node_pre.next = None
        return node.value

    def remove(self, index):
        """删除指定结点"""
        if (index < 0 or index >= self.length):
            raise ValueError('index is out of bound')
        if not self.head.next:
            raise ValueError('LinkedList id empty')
            
        node = self.head
        for _ in range(index):
            node = node.next
        # 此时 node 已经是 index 索引前面的一个结点（因为从哨兵结点开始）
        del_node = node.next
        node.next = node.next.next
        self.length -= 1
        return del_node
    
    def printlist(self):
        """打印链表"""
        node = self.head.next
        count = 0
        while node and count < 20:
            print(node.value, end=" ")
            node = node.next
            count += 1
        print('')
```

### 2. 找到中间节点

思路： 【双指针】

一个 2 倍速 一个 1 倍速，fast 到达终点时，slow 正好在一半

```python
def find_middle(linklis):
    assert linklis.head is not None and linklis.head.next is not None

    fast = linklis.head
    slow = linklis.head
    while fast is not None and fast.next is not None:
        fast = fast.next.next
        slow = slow.next
    # 此时 fast 到达终点，slow 正好在一半
    return slow.value
```

### 3. 判断链表是否有环

思路：判断 1 倍速结点与 2 倍速结点是否相遇
```python
def has_sycle(linklis):
    return has_cycle_helper(linklis.head)

def has_cycle_helper(head):
    if head is None:
        return False
    
    fast = head
    slow = head
    while fast is not None and fast.next is not None:
        fast = fast.next.next
        slow = slow.next
        if fast == slow:    # 相遇 有环（如果有环，两个结点都进入环之后最多再跑两圈救会相遇）
            return True
    return False            # fast 跑完到了尾结点（环没有尾结点）
```

### 4. 找到环的开始节点


```python
def find_beginning(head):
    if head is None:
        return False
    
    fast = head
    slow = head
    while fast is not None and fast.next is not None:
        fast = fast.next.next
        slow = slow.next
        if fast == slow:
            slow = head
            break
    if fast is None or fast.next is None:   # 有尾结点 无环
        return None
    # 此时 fast 在两节点环内相交的位置，slow 在链表起点的位置
    while fast != slow:
        fast = fast.next    # 双方都有一倍速接着走，在环起点的位置会相交
        slow = slow.next
    return slow
```

### 5. 删除倒数第N个节点


```python
def remove_nth(linklis, n):
    assert n > 0 and n <= linklis.length

    fast = linklis.head
    for _ in range(n):      # 从哨兵结点开始 到 n 位置的前一个几点
        fast = fast.next
    # 此时 fast 在 n 索引指向的位置 前一个结点(为了方便删除）
    slow = linklis.head
    while fast.next is not None:
        fast = fast.next
        slow = slow.next    # 双方都用同样的速度
    # 此时 slow 在倒数第 N 个结点的位置 前一个结点
    # 执行删除
    del_node = slow.next
    slow.next = slow.next.next
    linklis.length -= 1
    return del_node
```
